void pretty_print(const string& prefix, const Tree* root, bool isLeft, ostream& out) {
    if (root != nullptr) {
        out << prefix;
        out << (isLeft ? "├──" : "└──" );
        out << root->key << endl;
        print(prefix + (isLeft ? "│   " : "    "), root->left, true, out);
        print(prefix + (isLeft ? "│   " : "    "), root->right, false, out);
    }
}

void pretty_print(const Tree* root, ostream& out) {
    print("", root, false, out);
}
